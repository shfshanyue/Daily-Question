# 以下代码，koa 会返回什么数据

::: tip 更多描述 
 根据有无 `await next()` 判断以下代码输出结果

``` js
const Koa = require('koa')
const app = new Koa()

app.use(async (ctx, next) => {
  ctx.body = 'hello, 1'
})

app.use(ctx => {
  ctx.body = 'hello, 2'
})

app.listen(3000)
```

``` js
const Koa = require('koa')
const app = new Koa()

app.use(async (ctx, next) => {
  ctx.body = 'hello, 1'
  await next()
})

app.use(ctx => {
  ctx.body = 'hello, 2'
})

app.listen(3000)
```

``` js
const Koa = require('koa')
const app = new Koa()

app.use(async (ctx, next) => {
  await next()
  ctx.body = 'hello, 1'
})

app.use(ctx => {
  ctx.body = 'hello, 2'
})

app.listen(3000)
```

